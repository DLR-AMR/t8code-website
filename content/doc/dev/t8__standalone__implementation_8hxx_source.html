582  elements_are_family (t8_element_t *const *fam) noexcept
583  {
584 #if T8_ENABLE_DEBUG
585  const int num_siblings = element_get_num_siblings (fam[0]);
586  for (int isib = 0; isib < num_siblings; isib++) {
587  T8_ASSERT (element_is_valid (fam[isib]));
588  }
589 #endif

591  t8_standalone_element<TEclass> parent, compare;
592  /* Take the parent of the first element as baseline to compare against */
593  element_get_parent ((const t8_element_t *) fam[0], (t8_element_t *) &parent);
594  const int num_children = element_get_num_children ((const t8_element_t *) &parent);
595  for (int childid = 0; childid < num_children; childid++) {
596  /* check whether each element has the same parent */
597  element_get_parent ((const t8_element_t *) fam[childid], (t8_element_t *) &compare);
598  if (element_compare ((const t8_element_t *) &parent, (const t8_element_t *) &compare)) {
599  return 0;

603  /* Could be replaced by type comparison as level is already checked in parent comparison */
604  element_get_child ((const t8_element_t *) &parent, childid, (t8_element_t *) &compare);

1119  static constexpr void
1120  element_set_linear_id (t8_element_t *elem, const t8_element_level level, t8_linearidx_t id) noexcept
1121  {
1122 
1123  t8_standalone_element<TEclass> *el = (t8_standalone_element<TEclass> *) elem;
1124 
1125  set_to_root ((t8_element_t *) el);
1126 
1127  /* There is only one element at level 0, so it must be root */
1128  if (level == 0) {
1129  T8_ASSERT (id == 0);
1130  return;
1131  }
1132 
1133  T8_ASSERT (id < (size_t) element_count_leaves (elem, level));
1134  T8_ASSERT (1 <= level && level <= T8_ELEMENT_MAXLEVEL[TEclass]);
1135  t8_standalone_element<TEclass> child;
1136 
1137  while (el->level < level) {
1138  /* Shortcut if we need the first descendant of the subtree*/
1139  if (id == 0) {
1140  element_get_first_descendant ((const t8_element_t *) el, (t8_element_t *) el, level);
1141  return;
1142  }
1143 
1144  t8_linearidx_t sum_descendants_of_children_before;
1145  t8_linearidx_t sum_descendants_of_children_until_current = 0;
1146  int childindex = -1;
1147 
1148  /* Find the first child id so that the sum of descendants of previous child and the own number of descendants is greater than id */
1149  do {
1150  /* Go to the next child */
1151  sum_descendants_of_children_before = sum_descendants_of_children_until_current;
1152  childindex++;
1153  T8_ASSERT (childindex < element_get_num_children ((const t8_element_t *) el));
1154 
1155  element_get_num_children ((const t8_element_t *) el);
1156 
1157  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) &child);
1158  const t8_linearidx_t num_descendants_of_child = element_count_leaves ((t8_element_t *) &child, level);
1159 
1160  /* Add number of descendant of current child to cumulative sum */
1161  sum_descendants_of_children_until_current = sum_descendants_of_children_before + num_descendants_of_child;
1162 
1163  } while (sum_descendants_of_children_until_current <= id);
1164 
1165  /* Replace el by child to go into next iteration at finer level*/
1166  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) el);
1167  /* get id in subtree of child */
1168  id -= sum_descendants_of_children_before;
1169  }
1170  T8_ASSERT (id == 0);
1171  return;
1172  }
