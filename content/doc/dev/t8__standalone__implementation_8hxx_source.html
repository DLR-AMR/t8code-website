539  static constexpr int
540  elements_are_family (t8_element_t *const *fam) noexcept
541  {
542 #if T8_ENABLE_DEBUG
543  const int num_siblings = element_get_num_siblings (fam[0]);
544  for (int isib = 0; isib < num_siblings; isib++) {
545  T8_ASSERT (element_is_valid (fam[isib]));
546  }
547 #endif
548 
549  t8_standalone_element<TEclass> parent, compare;
550  /* Take the parent of the first element as baseline to compare against */
551  element_get_parent ((const t8_element_t *) fam[0], (t8_element_t *) &parent);
552  const int num_children = element_get_num_children ((const t8_element_t *) &parent);
553  for (int childid = 0; childid < num_children; childid++) {
554  /* check whether each element has the same parent */
555  element_get_parent ((const t8_element_t *) fam[childid], (t8_element_t *) &compare);
556  if (element_compare ((const t8_element_t *) &parent, (const t8_element_t *) &compare)) {
557  return 0;
558  }
559 
560  /* check whether each element is the correct child of the collective parent */
561  /* Could be replaced by type comparison as level is already checked in parent comparison */
562  element_get_child ((const t8_element_t *) &parent, childid, (t8_element_t *) &compare);
563 
564  if (element_compare ((const t8_element_t *) fam[childid], (const t8_element_t *) &compare)) {
565  return 0;
566  }
567  }
568  return 1;
569  }

902  static constexpr void
903  element_set_linear_id (t8_element_t *elem, const t8_element_level level, t8_linearidx_t id) noexcept
904  {
905 
906  t8_standalone_element<TEclass> *el = (t8_standalone_element<TEclass> *) elem;
907 
908  get_root ((t8_element_t *) el);
909 
910  /* There is only one element at level 0, so it must be root */
911  if (level == 0) {
912  T8_ASSERT (id == 0);
913  return;
914  }
915 
916  T8_ASSERT (0 <= id);
917  T8_ASSERT (1 <= level && level <= T8_ELEMENT_MAXLEVEL[TEclass]);
918  t8_standalone_element<TEclass> child;
919 
920  while (el->level < level) {
921  /* Shortcut if we need the first descendant of the subtree*/
922  if (id == 0) {
923  element_get_first_descendant ((const t8_element_t *) el, (t8_element_t *) el, level);
924  return;
925  }
926 
927  t8_linearidx_t sum_descendants_of_children_before;
928  t8_linearidx_t sum_descendants_of_children_until_current = 0;
929  int childindex = -1;
930 
931  /* Find the first child id so that the sum of descendants of previous child and the own number of descendants is greater than id */
932  do {
933  /* Go to the next child */
934  sum_descendants_of_children_before = sum_descendants_of_children_until_current;
935  childindex++;
936  T8_ASSERT (childindex < element_get_num_children ((const t8_element_t *) el));
937 
938  element_get_num_children ((const t8_element_t *) el);
939 
940  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) &child);
941  const t8_linearidx_t num_descendants_of_child = element_count_leaves ((t8_element_t *) &child, level);
942 
943  /* Add number of descendant of current child to cumulative sum */
944  sum_descendants_of_children_until_current = sum_descendants_of_children_before + num_descendants_of_child;
945 
946  } while (sum_descendants_of_children_until_current <= id);
947 
948  /* Replace el by child to go into next iteration at finer level*/
949  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) el);
950  /* get id in subtree of child */
951  id -= sum_descendants_of_children_before;
952  }
