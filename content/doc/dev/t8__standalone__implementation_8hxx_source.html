574  {
575 #if T8_ENABLE_DEBUG
576  const int num_siblings = element_get_num_siblings (fam[0]);
577  for (int isib = 0; isib < num_siblings; isib++) {
578  T8_ASSERT (element_is_valid (fam[isib]));
579  }
580 #endif
581 
582  t8_standalone_element<TEclass> parent, compare;
583  /* Take the parent of the first element as baseline to compare against */
584  element_get_parent ((const t8_element_t *) fam[0], (t8_element_t *) &parent);
585  const int num_children = element_get_num_children ((const t8_element_t *) &parent);
586  for (int childid = 0; childid < num_children; childid++) {
587  /* check whether each element has the same parent */
588  element_get_parent ((const t8_element_t *) fam[childid], (t8_element_t *) &compare);
589  if (element_compare ((const t8_element_t *) &parent, (const t8_element_t *) &compare)) {
590  return 0;

592

594  /* Could be replaced by type comparison as level is already checked in parent comparison */
595  element_get_child ((const t8_element_t *) &parent, childid, (t8_element_t *) &compare);

1110  static constexpr void
1111  element_set_linear_id (t8_element_t *elem, const t8_element_level level, t8_linearidx_t id) noexcept
1112  {
1113 
1114  t8_standalone_element<TEclass> *el = (t8_standalone_element<TEclass> *) elem;
1115 
1116  set_to_root ((t8_element_t *) el);
1117 
1118  /* There is only one element at level 0, so it must be root */
1119  if (level == 0) {
1120  T8_ASSERT (id == 0);
1121  return;
1122  }
1123 
1124  T8_ASSERT (id < (size_t) element_count_leaves (elem, level));
1125  T8_ASSERT (1 <= level && level <= T8_ELEMENT_MAXLEVEL[TEclass]);
1126  t8_standalone_element<TEclass> child;
1127 
1128  while (el->level < level) {
1129  /* Shortcut if we need the first descendant of the subtree*/
1130  if (id == 0) {
1131  element_get_first_descendant ((const t8_element_t *) el, (t8_element_t *) el, level);
1132  return;
1133  }
1134 
1135  t8_linearidx_t sum_descendants_of_children_before;
1136  t8_linearidx_t sum_descendants_of_children_until_current = 0;
1137  int childindex = -1;
1138 
1139  /* Find the first child id so that the sum of descendants of previous child and the own number of descendants is greater than id */
1140  do {
1141  /* Go to the next child */
1142  sum_descendants_of_children_before = sum_descendants_of_children_until_current;
1143  childindex++;
1144  T8_ASSERT (childindex < element_get_num_children ((const t8_element_t *) el));
1145 
1146  element_get_num_children ((const t8_element_t *) el);
1147 
1148  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) &child);
1149  const t8_linearidx_t num_descendants_of_child = element_count_leaves ((t8_element_t *) &child, level);
1150 
1151  /* Add number of descendant of current child to cumulative sum */
1152  sum_descendants_of_children_until_current = sum_descendants_of_children_before + num_descendants_of_child;
1153 
1154  } while (sum_descendants_of_children_until_current <= id);
1155 
1156  /* Replace el by child to go into next iteration at finer level*/
1157  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) el);
1158  /* get id in subtree of child */
1159  id -= sum_descendants_of_children_before;
1160  }
1161  T8_ASSERT (id == 0);
1162  return;
1163  }

1172  element_get_linear_id (const t8_element_t *elem, const t8_element_level level) noexcept
1173  {

1175  const t8_standalone_element<TEclass> *el = (const t8_standalone_element<TEclass> *) elem;
1176  t8_standalone_element<TEclass> ancestor;
1177
