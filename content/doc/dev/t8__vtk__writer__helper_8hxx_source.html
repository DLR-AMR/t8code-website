105 t8_forest_vtk_get_element_nodes (t8_forest_t forest, t8_locidx_t ltreeid, const t8_element_t *element, const int vertex,

257 grid_do_ghosts<t8_forest_t> (const t8_forest_t grid, const int write_ghosts)
258 {
259  bool ghosts = write_ghosts;
260  if (grid->ghosts == NULL || grid->ghosts->num_ghosts_elements == 0) {
261  /* Never write ghost elements if there aren't any */
262  ghosts = false;
263  }
264  T8_ASSERT (grid->ghosts != NULL || !ghosts);
265  return ghosts;
266 }
