270  private:
277  template <typename geometry_type>
278  inline geometry_type *
279  add_geometry (std::unique_ptr<t8_geometry> geom)
280  {
281  t8_debugf ("Registering geometry with name %s\n", geom->t8_geom_get_name ().c_str ());
282  const t8_geometry_hash hash = geom->t8_geom_get_hash ();
283  if (registered_geometries.find (hash) == registered_geometries.end ()) {
284  registered_geometries.emplace (hash, std::move (geom));
285  }
286  else {
287  t8_productionf ("WARNING: Did not register the geometry %s because it is already registered.\n"
288  "Geometries only need to be registered once per process.\n"
289  "If you are registering a new geometry it probably has the same name as another one.\n",
290  geom->t8_geom_get_name ().c_str ());
291  }
292  if (registered_geometries.size () == 1) {
293  active_geometry = registered_geometries.at (hash).get ();
294  }
295  return static_cast<geometry_type *> (registered_geometries.at (hash).get ());
296  }
