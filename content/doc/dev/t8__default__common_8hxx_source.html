79 /* Given an element's level and dimension, return the number of leaves it
80  * produces at a given uniform refinement level */
81 static inline t8_gloidx_t
82 count_leaves_from_level (const int element_level, const int refinement_level, const int dimension)
83 {
84  return element_level > refinement_level ? 0 : (1ULL << (dimension * (refinement_level - element_level)));
85 }
